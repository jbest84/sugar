Sugar - Dojo AMD plugin for loading coffeescript

CoffeeScript Module:

define('some/coffeescript/module', [], () ->
  module = 
    someFunc : (arg) -> arg + 1
  )

In your javascript:

require(['sugar/cs!./some/coffeescript/module'], function(module) {
  // Loads your coffeescript files
  module.someFunc(100);
});
